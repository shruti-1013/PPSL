#include <iostream>
using namespace std;

//base class vehicle
class Vehicle
{
  public :
  void vehicle()
  {
    cout<<"I am a vehicle"<<endl;
  }
};

//derived class two wheeler
class Twowheeler : public Vehicle
{
  public :
  void twowheeler()
  {
    cout<<"I have two wheels"<<endl;
  }
};

//derived class for bike
class Bike : public Twowheeler
{
  public :
  void bike()
  {
    cout<<"I am a bike"<<endl;
  }
};

int main()
{
//create an object of class bike
Bike b2;
b2.bike();
b2.twowheleer();
b2.vehicle();

return 0;
}
